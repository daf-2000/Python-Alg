#4. Написать программу, которая генерирует в указанных пользователем границах:
#случайное целое число;
#cлучайное вещественное число;
#случайный символ. Для каждого из трех случаев пользователь задает свои границы диапазона. Например, если надо получить случайный символ от 'a' до 'f', то вводятся эти символы. Программа должна вывести на экран любой символ алфавита от 'a' до 'f' включительно.

import random
x1 = int(input('Введите первое число для диапазона целых чисел'))
x2 = int(input('Введите второе число для диапазона целых чисел'))
y1 = float(input('Введите первое число для диапазона вещественных чисел'))
y2 = float(input('Введите второе число для диапазона вещественных чисел'))
z1 = input('Введите первый символ диапазона')
z2 = input('Введите второй символ диапазона')
#Для int
if x1 < x2:
    print(f'Случайное целое число из указанного диапазона {random.randint( x1, x2 )}')
elif x1 > x2:
    print(f'Случайное целое число из указанного диапазона {random.randint( x2, x1 )}')
else:
    print('Случайное целое число из указанного диапазона {x1}')
#Для Float
if ord(z1) < ord(z2):
    print(f'Случайное вещественное число из указанного диапазона {random.uniform( y1, y2 )}')
elif x1 > x2:
    print(f'Случайное вещественное число из указанного диапазона {random.uniform( y2, y1 )}')
else:
    print('Случайное вещественное число из указанного диапазона {y1}')
#Для str
if ord(z1) < ord(z2):
    rand = random.randint(ord(z1), ord(z2))
    print(f'Случайный символ из указанного диапазона = {chr(rand)}')
elif ord(z2) < ord(z1):
    rand = random.randint(ord(z2), ord(z1))
    print(f'Случайный символ из указанного диапазона = {chr(rand)}')
else:
    print(f'Случайный символ из указанного диапазона = {z1}')
