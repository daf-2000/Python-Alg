#5. Пользователь вводит две буквы. Определить, на каких местах алфавита они стоят и сколько между ними находится букв.

import string
print('Введите 2 буквы')
x1 = input('x1= ')
x2 = input('x2= ')
al = string.ascii_lowercase
y1 = al.find(x1.lower())
y2 = al.find(x2.lower())
if y1 == y2:
    print(f"""Введены одинаковые буквы. Номер в алфавите {y1 + 1} Между ними 0 букв""")
else:
    print(f"Номер в алфавите буквы {x1} = {y1 + 1}, номер буквы {x2} = {y2 + 1}, Между ними {abs(y2 - y1) - 1} букв")
