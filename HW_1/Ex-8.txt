#8. Определить, является ли год, который ввел пользователем, високосным или невисокосным.

print('Введите год')
year = int(input('year= '))
if year % 400 == 0:
    print('Год високосный')
elif year % 100 == 0:
    print('Год НЕвисокосный')
elif year % 4 == 0:
    print('Год високосный')
else:
    print('Год НЕвисокосный')
